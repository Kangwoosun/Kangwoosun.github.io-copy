# Introduction

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>

int main(int argc, char* argv[]){
	char fname[128];
	unsigned long long otp[2];

	if(argc!=2){
		printf("usage : ./otp [passcode]\n");
		return 0;
	}

	int fd = open("/dev/urandom", O_RDONLY);

	if(fd==-1) 
		exit(-1);

	if(read(fd, otp, 16)!=16) 
		exit(-1);
	
	close(fd);

	sprintf(fname, "/tmp/%llu", otp[0]);
	
	FILE* fp = fopen(fname, "w");
	if(fp==NULL)
		{ exit(-1); }
	
	fwrite(&otp[1], 8, 1, fp);
	
	fclose(fp);

	printf("OTP generated.\n"); // fname file overwrite to passcode via python like watch dog

	unsigned long long passcode=0;
	
	FILE* fp2 = fopen(fname, "r"); // fopen overwritten file
	
	if(fp2==NULL)
		{ exit(-1); }
	
	fread(&passcode, 8, 1, fp2);
	
	fclose(fp2);
	
	if(strtoul(argv[1], 0, 16) == passcode){
		printf("Congratz!\n");
		system("/bin/cat flag");
	}
	else{
		printf("OTP mismatch\n");
	}

	unlink(fname);
	return 0;
}
```

# Vunlnerability

~~예상되는 취약점은 ~~

```c
	fwrite(&otp[1], 8, 1, fp);
	
	fclose(fp);

	printf("OTP generated.\n"); // fname file overwrite to passcode via python like watch dog

	unsigned long long passcode=0;
	
	FILE* fp2 = fopen(fname, "r"); // fopen overwritten file
```
~~에서 `fname`의 이름으로 만들어진 파일을 `fclose`를 했을때 즉 `OTP generated.\n`가 출력되는 순간 파이썬으로 새로 생성된 파일을 찾아서 원하는 passcode로 덮어씌우면 flag를 얻을 수 있을것같다.~~

정정... hint에 race condition 문제가 아니라고 되어있다..

찾아보니까 `ulimit`로 또 변태짓 하는 문제였다. `ulimit -f 0`을 주면 file size의 제한이 0이 되고 `fwrite`에서 에러가 발생한다.

```
Program received signal SIGXFSZ, File size limit exceeded.
0x00007ffff7b042c0 in __write_nocancel () at ../sysdeps/unix/syscall-template.S:84
84      ../sysdeps/unix/syscall-template.S: No such file or directory.
```

에러문을 보면 `SIGXFSZ`시그널을 받게되는데 gdb로 시그널을 무시하고 실행하면 

```c
	fread(&passcode, 8, 1, fp2);
	
	fclose(fp2);
	
	if(strtoul(argv[1], 0, 16) == passcode){
		printf("Congratz!\n");
		system("/bin/cat flag");
	}
	else{
		printf("OTP mismatch\n");
	}

	unlink(fname);
	return 0;
```

`fread(&passcode, 8, 1, fp2)`에서 `passcode`에 0이 들어가기 때문에 위에서 나온 `SIGXFSZ` 시그널을 무시하는 코드를 짜고 `argv[1]`에 0을 넣어주면 문제가 풀릴거라 예상한다.
